[docs]def vec2full(arr):
    shp=arr.shape
    n=int((np.sqrt(8*shp[0]+1)-1)/2)
    oshp=[n,n]+list(shp[1:])
    g=np.zeros(oshp,dtype=arr.dtype)
    indx=0
    for jj in range(n):
        for ii in range(jj,n):
            if ii==jj:
                g[ii,jj]=arr[indx]
            else:
                g[ii,jj]=arr[indx]
                g[jj,ii]=arr[indx]
            indx+=1
    return g
